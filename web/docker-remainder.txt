version: '3'

services:
  database:
    container_name: cube_planning_mysql_db
    image: mysql:5.6
    environment:
      - MYSQL_DATABASE=cube_planning
      - MYSQL_USER=web
      - MYSQL_PASSWORD=root
      - MYSQL_ROOT_PASSWORD=root
    ports:
      - "127.0.0.1:3310:3306"
    volumes:
      - mysqldata:/var/lib/mysql
    networks:
      - cube_network

  redis:
    container_name: cube_planning_redis
    image: redis:5.0.3
    ports:
      - "6379:6379"
    networks:
      - cube_network

  web:
    build: .
    container_name: cube_planning
    command: python3 manage.py runserver 0.0.0.0:8000
    env_file:
      - .env
    volumes:
      - .:/src
      - ~/.aws:/.aws
    ports:
      - "8000:8000"
    depends_on:
      - database
    networks:
      - cube_network

  worker:
    build: .
    container_name: cube_planning_worker
    command: python manage.py celery_dev
    image: &img worker
    env_file:
      - .env
    volumes:
      - .:/src
      - ~/.aws:/.aws
    depends_on:
      - database
      - redis
      - web
    networks:
      - cube_network

  flower:
    container_name: cube_planning_flower
    image: mher/flower
    env_file:
      - .env
    ports:
      - "5555:5555"
    depends_on:
      - redis
    networks:
      - cube_network

volumes:
  mysqldata:

networks:
  cube_network:
    driver: bridge
